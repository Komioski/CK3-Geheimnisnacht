coa_theocratic_trigger = {
	exists = root
	OR = {
		government_has_flag = government_is_theocracy
		AND = {
			exists = liege
			liege = {
				any_sub_realm_barony = {
					lessee = ROOT
					title_province = {
						has_holding_type = church_holding
					}
				}
			}
		}
	}
}
